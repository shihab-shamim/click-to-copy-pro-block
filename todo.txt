	// ---------------- Plugin Class ----------------
	class CTCB_ClickToCopy {
		private static $instance;

		private function __construct() {
			$this->constants_define();
			add_action( 'init', [ $this, 'onInit' ] );
             // freemius
                  add_action('wp_ajax_bpicbPremiumChecker', [$this, 'bpicbPremiumChecker']);
	 				add_action('wp_ajax_nopriv_bpicbPremiumChecker', [$this, 'bpicbPremiumChecker']);

            function bpicbPremiumChecker(){
					$nonce = sanitize_text_field($_POST['_wpnonce'] ?? null);
		
					if (!wp_verify_nonce($nonce, 'wp_ajax')) {
						wp_send_json_error('Invalid Request');
					}
		
					wp_send_json_success([
						'isPipe' => ctcIsPremium()
						
					
						
					]);
				}
		
				function registerSettings(){

						'sanitize_callback' => 'sanitize_text_field'
					]);
				}


		public static function get_instance() {
			if ( self::$instance ) {
				return self::$instance;
			}

			return self::$instance;
		}

		private function constants_define() {
			define( 'CTCB_PLUGIN_VERSION', isset( $_SERVER['HTTP_HOST'] ) && 'localhost' === $_SERVER['HTTP_HOST'] ? time() : '1.0.4' );

			define( 'CTCB_DIR_PATH', plugin_dir_path( __FILE__ ) );
			define( 'CTCB_ASSETS_DIR', plugin_dir_url( __FILE__ ) . 'assets/' );
		}

		public function onInit() {
			register_block_type( __DIR__ . '/build' );

			wp_set_script_translations( 'ctcb-click-to-copy-editor-script', 'clipboard', plugin_dir_path( __FILE__ ) . 'languages' );
